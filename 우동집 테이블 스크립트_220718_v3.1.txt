-- 개인회원 --

CREATE TABLE MEMBER (
	MEMBER_NO NUMBER PRIMARY KEY,
	MEMBER_ID VARCHAR2(18) NOT NULL UNIQUE,
	MEMBER_PWD VARCHAR2(100) NOT NULL,
	MEMBER_NAME VARCHAR2(18) NOT NULL,
	MEMBER_PHONE VARCHAR2(11) NOT NULL,
	MEMBER_EMAIL VARCHAR2(40) NOT NULL,
	STATUS VARCHAR2(1) DEFAULT 'Y' CHECK (STATUS IN ('Y', 'N'))
);

COMMENT ON COLUMN MEMBER.MEMBER_NO IS '개인회원번호';
COMMENT ON COLUMN MEMBER.MEMBER_ID IS '개인회원아이디';

COMMENT ON COLUMN MEMBER.MEMBER_NAME IS '개인회원이름';
COMMENT ON COLUMN MEMBER.MEMBER_PHONE IS '개인회원연락처';
COMMENT ON COLUMN MEMBER.MEMBER_EMAIL IS '개인회원이메일';

-- 매물 --

CREATE TABLE HOUSE (
	HOUSE_NO NUMBER PRIMARY KEY,
	AGENT_NO NUMBER NOT NULL,
	TITLE VARCHAR2(300) NOT NULL,
	DESCRIPTION VARCHAR2(3000) NOT NULL,
	THUMBNAIL VARCHAR2(300),
	SALES_TYPE VARCHAR2(20) NOT NULL,
	ADDRESS1 VARCHAR2(100) NOT NULL,
	ADDRESS2 VARCHAR2(100) NOT NULL,
	ZIP_CODE VARCHAR2(5) NOT NULL,
	BUILDING_TYPE VARCHAR2(100) NOT NULL,
	BUILDING_FLOOR VARCHAR2(20) NOT NULL,
	SUBWAY_NO NUMBER NOT NULL,
	DEPOSIT NUMBER NOT NULL,
	MONTHLY_COST NUMBER DEFAULT 0 NOT NULL,
	MANAGE_COST NUMBER NOT NULL,
	FLOOR VARCHAR2(20) NOT NULL,
	ROOM_TYPE VARCHAR2(100) NOT NULL,
	DIRECTION VARCHAR2(15) NOT NULL,
	PARKING NUMBER DEFAULT 0 NOT NULL,
	LAT VARCHAR2(30) NOT NULL,
	LNG VARCHAR2(30) NOT NULL,
	SIZE_M2 NUMBER NOT NULL,
	SIZE_P NUMBER NOT NULL,
	PET VARCHAR2(6) DEFAULT '불가' CHECK (PET IN ('가능', '불가')) NOT NULL ,
	LOAN VARCHAR2(6) DEFAULT '불가' CHECK (LOAN IN ('가능', '불가')) NOT NULL ,
    STATUS VARCHAR2(1) DEFAULT 'Y' CHECK (STATUS IN ('Y', 'N')),
	MOVEIN_DATE VARCHAR2(30) NOT NULL,
	REG_DATE DATE DEFAULT SYSDATE NOT NULL ,
	VIEW_COUNT NUMBER DEFAULT 0 NOT NULL ,
	REPORT_COUNT NUMBER DEFAULT 0 NOT NULL ,
	SALES_STATUS VARCHAR2(1) DEFAULT 'Y' CHECK (SALES_STATUS IN ('Y', 'N')) NOT NULL,
    FOREIGN KEY(AGENT_NO) REFERENCES AGENT (AGENT_NO),
    FOREIGN KEY(SUBWAY_NO) REFERENCES SUBWAY (SUBWAY_NO)
);

COMMENT ON COLUMN HOUSE.HOUSE_NO IS '매물 번호';

COMMENT ON COLUMN HOUSE.SALES_TYPE IS '계약유형(월세/전세)';
COMMENT ON COLUMN HOUSE.ADDRESS1 IS '시/군/구/도로명';
COMMENT ON COLUMN HOUSE.ADDRESS2 IS '상세주소';
COMMENT ON COLUMN HOUSE.ZIP_CODE IS '우편번호';
COMMENT ON COLUMN HOUSE.BUILDING_TYPE IS '건물 유형 (다가구/오피스텔 등)';
COMMENT ON COLUMN HOUSE.BUILDING_FLOOR IS '건물 전체 층수';
COMMENT ON COLUMN HOUSE.SUBWAY_NO IS '가장 가까운 지하철역 정보';
COMMENT ON COLUMN HOUSE.DEPOSIT IS '월세 : 보증금 / 전세 : 전세금';

* 오타 수정 및 누락 구문 추가

* SUBWAY 테이블
  - SUBWAY_NAME 컬럼명 STATION으로 변경

* HOUSE 테이블
  - THUMBNAIL 컬럼 NOT NULL 제약조건 삭제 (INSERT 과정에서 순서 문제로 인해 삭제)
  - ROOM_COUNT 컬럼 삭제 (ROOM_TYPE 컬럼에 도메인이 '오픈형 원룸 / 분리형 원룸 / 복층형 원룸 / 투룸'으로 구분 예정)
  - MOVEIN_DATE 컬럼 자료형 DATE => VARCHAR2(30)으로 변경 ('입주가능한 날짜' 또는 '즉시입주' 문자열이 들어갈 예정)
  - PET, LOAN 컬럼 DEFAULT '불가', CHECK 제약조건 (컬럼명 IN '가능', '불가')로 변경 (값 출력 시 편의를 위해 변경)

AGENT 테이블
- COMPANY_NO 컬럼 NUMBER => VARCHAR2(10)

HOUSE 테이블
- ZIP_CODE 컬럼 NUMBER => VARCHAR2(5)
- LAT / LNG 컬럼 NUMBER => VARCHAR2(30)

NOTICE 테이블
- TITLE 컬럼 VARCHAR2(60) => VARCHAR2(500)

* 제약조건 삭제

CHATMESSAGE 테이블
- UNREAD 컬럼 DEFAULT 'N' 삭제
